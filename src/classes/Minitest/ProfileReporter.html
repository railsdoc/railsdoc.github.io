def record(result)
  if output_file = ENV["RAILTIES_OUTPUT_FILE"]
    File.open(output_file, "a") do |f|
      # Round-trip for re-serialization
      data = JSON.parse(result.to_json)
      data[:location] = result.location
      f.puts(data.to_json)
    end
  else
    @results << result
  end
end
