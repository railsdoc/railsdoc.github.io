def fetch_entry(key)
  entry = super do
    new_entry = yield
    if entry.is_a?(Entry)
      new_entry.dup_value!
    end
    new_entry
  end
  entry = entry.dup

  if entry.is_a?(Entry)
    entry.dup_value!
  end

  entry
end
