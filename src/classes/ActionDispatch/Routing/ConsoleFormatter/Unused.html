def no_routes(routes, filter)
  @buffer <<
    if filter.none?
      "No unused routes found."
    elsif filter.key?(:controller)
      "No unused routes found for this controller."
    elsif filter.key?(:grep)
      "No unused routes found for this grep pattern."
    end
end
