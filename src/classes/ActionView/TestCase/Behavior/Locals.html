def render(options = {}, local_assigns = {})
  case options
  when Hash
    if block_given?
      rendered_views.add options[:layout], options[:locals]
    elsif options.key?(:partial)
      rendered_views.add options[:partial], options[:locals]
    end
  else
    rendered_views.add options, local_assigns
  end

  super
end
