primary_key=(value)

Sets the name of the primary key column.

class Project < ActiveRecord::Base
  self.primary_key = 'sysid'
end

You can also define the primary_key method yourself:

class Project < ActiveRecord::Base
  def self.primary_key
    'foo_' + super
  end
end

Project.primary_key # => "foo_id"
