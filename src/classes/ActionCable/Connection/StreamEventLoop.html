def attach(io, stream)
  @todo << lambda do
    @map[io] = @nio.register(io, :r)
    @map[io].value = stream
  end
  wakeup
end

def detach(io, stream)
  @todo << lambda do
    @nio.deregister io
    @map.delete io
    io.close
  end
  wakeup
end

def post(task = nil, &block)
  task ||= block

  spawn
  @executor << task
end

def writes_pending(io)
  @todo << lambda do
    if monitor = @map[io]
      monitor.interests = :rw
    end
  end
  wakeup
end
