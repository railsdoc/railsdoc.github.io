def self.generate_method_for_mime(mime)
  sym = mime.is_a?(Symbol) ? mime : mime.to_sym
  class_eval <<-RUBY, __FILE__, __LINE__ + 1
    def #{sym}(...)
      custom(Mime[:#{sym}], ...)
    end
  RUBY
end
