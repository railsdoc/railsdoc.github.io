    def initialize
      super
      @root = "#{Rails.root}/"
      add_filter { |line| line.sub(@root, EMPTY_STRING) }
      add_filter { |line| line.sub(RENDER_TEMPLATE_PATTERN, EMPTY_STRING) }
      add_filter { |line| line.sub(DOT_SLASH, SLASH) } # for tests
      add_silencer { |line| !APP_DIRS_PATTERN.match?(line) }
    end
