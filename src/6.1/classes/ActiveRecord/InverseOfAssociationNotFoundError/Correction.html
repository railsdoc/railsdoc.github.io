def corrections
  if @error.reflection && @error.associated_class
    maybe_these = @error.associated_class.reflections.keys

    maybe_these.sort_by { |n|
      DidYouMean::Jaro.distance(@error.reflection.options[:inverse_of].to_s, n)
    }.reverse.first(4)
  else
    []
  end
end
