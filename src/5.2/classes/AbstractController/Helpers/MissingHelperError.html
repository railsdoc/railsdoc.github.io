      def initialize(error, path)
        @error = error
        @path  = "helpers/#{path}.rb"
        set_backtrace error.backtrace

        if error.path =~ /^#{path}(\.rb)?$/
          super("Missing helper file helpers/%s.rb" % path)
        else
          raise error
        end
      end
